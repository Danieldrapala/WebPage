Gra Chinese Chekers

Projekt realizowany na uczelnie. Chinese Checkers to gra bardzo fajna. Jest grana przez ludzi z Chin(stąd nazwa), dlatego razem z kolegą postanowiliśmy ją stworzyć


Ponizej znajduje sie jedna funkcja z kodu ktory pokazuje mozliwe ruchy dla przycisnietego pionka. 

    public void showPossbileMoves(Field selected2) {
      if (!selected2.getColor().equals("WHITE")) {
        int ySelected = selected2.getYCord();
        int xSelected = selected2.getXCord();
                        
        if (ySelected % 2 == 1) {
          highlightField(this.getNode(ySelected - 1, xSelected - 1));
          highlightField(this.getNode(ySelected + 1, xSelected - 1));
        } else {
          highlightField(this.getNode(ySelected - 1, xSelected + 1));
          highlightField(this.getNode(ySelected + 1, xSelected + 1));
        }
        highlightField(this.getNode(ySelected - 1, xSelected));
        highlightField(this.getNode(ySelected + 1, xSelected));
        highlightField(this.getNode(ySelected, xSelected - 1));
        highlightField(this.getNode(ySelected, xSelected + 1));
        findAHop(selected2);		
  }
